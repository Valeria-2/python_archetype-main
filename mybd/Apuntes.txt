Subtitulo: Script Para base de datos de ejemplo

CREATE TABLE `tbl_role` (
  `cd_id` int NOT NULL AUTO_INCREMENT,
  `nb_name` varchar(100) NOT NULL,
  `nb_usercreated` varchar(100) NOT NULL,
  `ts_created` datetime NOT NULL,
  `nb_usermodified` varchar(100) NOT NULL,
  `ts_modified` datetime NOT NULL,
  `st_active` tinyint NOT NULL DEFAULT '1',
  PRIMARY KEY (`cd_id`)
);


CREATE TABLE `tbl_user` (
  `cd_id` int NOT NULL AUTO_INCREMENT,
  `cd_role` int NOT NULL,
  `nb_name` varchar(100) NOT NULL,
  `nb_username` varchar(100) NOT NULL,
  `nb_email` varchar(100) NOT NULL,
  `nb_usercreated` varchar(100) NOT NULL,
  `ts_created` datetime NOT NULL,
  `nb_usermodified` varchar(100) NOT NULL,
  `ts_modified` datetime NOT NULL,
  `st_active` tinyint NOT NULL DEFAULT '1',
  PRIMARY KEY (`cd_id`),
  FOREIGN KEY(cd_role) REFERENCES tbl_role(cd_id)
);
